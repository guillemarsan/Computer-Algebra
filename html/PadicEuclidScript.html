a = 48;

p = 11;

Qp = Padics(p);

fprintf("a = %s\n", Qp.padicsshow(Qp,a));
fprintf("b = %s\n", Qp.padicsshow(Qp,b));
gcd = euclid(a,b,Qp); % -3
fprintf("gcd(a,b) = %s\n",Qp.padicsshow(Qp,gcd));

b = (2)*11^0 + (2)*11^1
gcd(a,b) = (2)*11^0 + (2)*11^1
