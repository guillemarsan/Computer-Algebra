MCD en un DFU

Cálculo de con y un cualquiera

path(path, 'Algorithms');
path(path, 'Classes');
clear


f = [1 10 23 14];
g = [1 0 -19 -30];


Zx = IntegerPoly;

fprintf("f(x) = %s\n",Zx.polyshow(Zx,f));
fprintf("g(x) = %s\n",Zx.polyshow(Zx,g));

gcd = euclid_primitive(f,g,Zx); % x+2
fprintf("gcd(f(x),g(x)) = %s\n", Zx.polyshow(Zx,gcd));

f(x) = 1x^3+10x^2+23x+14
g(x) = 1x^3+0x^2-19x-30
gcd(f(x),g(x)) = 1x+2


f = [1 7 9 13];
g = [-2 -10 10 10 52 0];

Zx = IntegerPoly;

fprintf("f(x) = %s\n",Zx.polyshow(Zx,f));
fprintf("g(x) = %s\n",Zx.polyshow(Zx,g));

gcd = euclid_primitive(f,g,Zx); % 1x^3+7x^2+9x+13
fprintf("gcd(f(x),g(x)) = %s\n", Zx.polyshow(Zx,gcd));

f(x) = 1x^3+7x^2+9x+13
g(x) = -2x^5-10x^4+10x^3+10x^2+52x+0
gcd(f(x),g(x)) = 1x^3+7x^2+9x+13


f = [1 7 9 13];
g = [2 3 5 -10 5];

Zx = IntegerPoly;

fprintf("f(x) = %s\n",Zx.polyshow(Zx,f));
fprintf("g(x) = %s\n",Zx.polyshow(Zx,g));

gcd = euclid_primitive(f,g,Zx); % 1
fprintf("gcd(f(x),g(x)) = %s\n", Zx.polyshow(Zx,gcd));

f(x) = 1x^3+7x^2+9x+13
g(x) = 2x^4+3x^3+5x^2-10x+5
gcd(f(x),g(x)) = 1
